<script>

const el = id => document.getElementById(id)

function showTab(i){
    document.querySelectorAll('.tab').forEach((t,idx)=>{
        t.classList.toggle('active', idx===i)
    })
    document.querySelectorAll('.section').forEach((s,idx)=>{
        s.classList.toggle('active', idx===i)
    })
}

function updateForm(){
    let s = el("shape").value

    el("rect").style.display = s==='rectangle'?'block':'none'
    el("circle").style.display = s==='circle'?'block':'none'
    el("cyl").style.display = s==='cylinder'?'block':'none'
}

function togglePosts(){
    el("manualPosts").style.display = el("autoPosts").checked ? 'none' : 'block'
}

function calc(){
    let s = el("shape").value
    let stepV = +el("step").value || 1
    let weave = +el("weaving").value || 1
    let thick = +el("thickness").value || 0
    let reserveV = (+el("reserve").value || 0)/100
    let priceV = +el("price").value || 0

    let total = 0
    let base = 0

    if(s==='rectangle'){
        let L = +el("length").value || 0
        let W = +el("width").value || 0
        let rows = W/stepV
        total = rows * L
        base = W
    }
    else if(s==='circle'){
        let D = +el("diameter").value || 0
        let C = Math.PI * D
        let rows = D/stepV
        total = rows * C
        base = D
    }
    else{
        let D = +el("cylD").value || 0
        let H = +el("height").value || 0
        let C = Math.PI * D
        let rows = H/stepV
        total = rows * C
        base = C
    }

    let spacingV = +el("spacing").value || 1
    let postH = +el("postH").value || 0

    let postsCount = el("autoPosts").checked
        ? Math.round(base/spacingV)
        : (+el("posts").value || 0)

    let gaps = Math.max(0, postsCount-1)
    let postsLen = (postsCount*postH)+(spacingV*gaps)

    let coef = 1+(thick/20)

    let weavingPart = total*weave*coef
    let postsPart = postsLen

    let sum = (weavingPart + postsPart)/100
    sum = sum*(1+reserveV)

    // ✅ ОКРУГЛЕНИЕ ДО СОТЕН
    let cost = Math.round((sum * priceV) / 100) * 100

    el("result").innerHTML = `
    <b>ИТОГ:</b><br><br>

    Плетение: ${(weavingPart/100).toFixed(2)} м<br>
    Стойки: ${(postsPart/100).toFixed(2)} м<br><br>

    <b>Общий расход:</b> ${sum.toFixed(2)} м<br>
    <b>Стоимость:</b> ${cost.toLocaleString('ru-RU')} ₽<br><br>

    <div class="small">
    Стоек: ${postsCount} шт<br>
    Коэф плетения: ${weave}<br>
    Толщина коэф: ${coef.toFixed(2)}
    </div>
    `
}
